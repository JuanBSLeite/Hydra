79  __host__ __device__ GReal_t pdk(const GReal_t a, const GReal_t b,
80  const GReal_t c) const
81  {
82  //the PDK function
83  GReal_t x = (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c);
84  x = sqrt(x) / (2 * a);

88  __host__ __device__ void bbsort(GReal_t *array, GInt_t n)
89  {
90  // Improved bubble sort
91  for (GInt_t c = 0; c < n; c++)
92  {
93  GInt_t nswap = 0;
94 
95  for (GInt_t d = 0; d < n - c - 1; d++)
96  {
97  if (array[d] > array[d + 1]) /* For decreasing order use < */
98  {
99  GReal_t swap = array[d];
100  array[d] = array[d + 1];
101  array[d + 1] = swap;
102  nswap++;
103  }
104  }
105  if (nswap == 0)
106  break;
107  }
