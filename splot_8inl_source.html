125  //gaussian
126  auto GAUSSIAN = [=] __host__ __device__
127  (unsigned int npar, const hydra::Parameter* params,unsigned int narg, double* x )
128  {
129  double m2 = (x[0] - params[0])*(x[0] - params[0] );
130  double s2 = params[1]*params[1];
131  double g = exp(-m2/(2.0 * s2 ))/( sqrt(2.0*s2*PI));

133  return g;
