66 /**
67  * @class
68  * @ingroup fit
69  * @brief Build a pdf adding other pdfs.
70  *
71  * Given N unnormalized pdfs \f$F_i\f$ , this class define a object representing the sum
72  * \f[ F_t = \sum_i^N c_i \times F_i \f]
73  * The coefficients of the pdfs can represent fractions or yields. If the number of coefficients is equal to
74  * the number of pdfs, the coefficients are interpreted as yields. If the number of coefficients is \f$(N-1)\f$,
75  * the coefficients are interpreted as fractions defined in the interval [0,1].
76  * The coefficient of the last term is calculated as \f$ c_N=1 -\sum_i^{(N-1)} c_i \f$.
77  */
78 template<typename PDF1, typename PDF2, typename ...PDFs>
79 class PDFSumExtendable: public detail::AddPdfBase<PDF1,PDF2,PDFs...>

103  /**
104  * \brief Ctor for used to build AddPdf usable in extended likelihood fits.
105  * \param pdf1 first pdf object,
106  * \param pdf2 second pdf object,
107  * \param pdfs remaining pdfs.
108  * \param coef arrary of Parameters, each parameter correspond to a coefficient
109  * \param extend build pdf to be used in extended fit. Default is true.
110  *
111  * Each component pdf is normalized properly before evaluation each time SetParameters(const std::vector<double>& parameters) is called.
112  * The sum is normalized also.
113  */
114  PDFSumExtendable( PDF1 const& pdf1, PDF2 const& pdf2, PDFs const& ...pdfs,
115  std::array<Parameter*, npdfs>const& coef ):
116  fPDFs(HYDRA_EXTERNAL_NS::thrust::make_tuple(pdf1,pdf2,pdfs...) ),
117  fFunctors(HYDRA_EXTERNAL_NS::thrust::make_tuple(pdf1.GetFunctor(),pdf2.GetFunctor(),pdfs.GetFunctor() ...) ),
118  fExtended(kTrue),
119  fCoefSum(0.0)
120  {
121  size_t i=0;
122  for(Parameter* var:coef){
123  fCoeficients[i] = *var;
124  fCoefSum += var->GetValue();
125  i++;
126  }
127 
128  }

166  /**

172  inline void SetParameters(const std::vector<double>& parameters){
173

175  fCoeficients[i].Reset(parameters );

185  }

266  {
