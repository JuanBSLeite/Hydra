120  * @group examples
121  *
122  * This example demonstrates how to generate Monte Carlo samples in parallel
123  * deploying different backends asynchronously.
124  */
125 
126 int main(int argv, char** argc)
127 {
128  size_t nentries = 0;
129 
130  try {
131 
132  TCLAP::CmdLine cmd("Command line arguments for ", '=');
133 
134  TCLAP::ValueArg<size_t> EArg("n", "number-of-events","Number of events", true, 10e6, "size_t");
135  cmd.add(EArg);
136 
137  // Parse the argv array.
138  cmd.parse(argv, argc);
139 
140  // Get the value parsed by each arg.
141  nentries = EArg.getValue();
142 
143  }
144  catch (TCLAP::ArgException &e) {
145  std::cerr << "error: " << e.error() << " for arg " << e.argId()
146  << std::endl;
147  }
148 
149  // some definitions
150  double min = -6.0;
151  double max = 6.0;
152  double mean = 0.0;
153  double sigma = 1.5;
154  size_t nbins = 100;
155 
156  auto GAUSSIAN3D = [=] __host__ __device__ (unsigned int n,double* x ){
157 
158  double g = 1.0;
159 
160  for(size_t i=0; i<3; i++){
161 
162  double m2 = (x[i] - mean )*(x[i] - mean );
163  double s2 = (sigma+i/2.0)*(sigma+i/2.0);
164  g *= exp(-m2/(2.0 * s2 ))/( sqrt(2.0*s2*PI));
165  }
166 
167  return g;
168  };
169 
170  auto Gaussian3D = hydra::wrap_lambda( GAUSSIAN3D );
171 
172

174  //generator
175  hydra::Random<>
176  Generator( std::chrono::system_clock::now().time_since_epoch().count() );
177

184  nbins, min, max,
185  nbins, min, max,
186  nbins, min, max );
187 
188  TH1D hist_omp("hist_omp", "Gaussian - OMP", nbins, min, max);
189 
190  TH1D hist_tbb("hist_tbb", "Breit-Wigner - TBB", nbins, min, max);
191 
192  TH1D hist_cpp("hist_cpp", "Uniform - CPP", nbins, min, max);
193 
194 #endif //_ROOT_AVAILABLE_

229  auto Histogram_CUDA = std::async( std::launch::async, [=, &dataset_gpu, &Generator] {
230 
231  std::array<double, 3> _max;

233  std::array<size_t, 3> _nbins;
234 
235  for(size_t i=0;i<3;i++){
236  _max[i] = max;
237  _min[i] = min;
238  _nbins[i] = nbins;
239  }
240 
241  auto range = Generator.Sample( dataset_gpu.begin(), dataset_gpu.end(), _min, _max, Gaussian3D);
242  hydra::SparseHistogram<3, double> Histogram(_nbins, _min, _max);
243  Histogram.Fill( range.begin(), range.end() );
