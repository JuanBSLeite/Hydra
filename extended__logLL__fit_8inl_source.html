120  //gaussian 1
121  auto GAUSSIAN1 = [=] __host__ __device__
122  (unsigned int npar, const hydra::Parameter* params,unsigned int narg, double* x )
123  {
124  double m2 = (x[0] - params[0])*(x[0] - params[0] );
125  double s2 = params[1]*params[1];
126  double g = exp(-m2/(2.0 * s2 ))/( sqrt(2.0*s2*PI));
127 
128  return g;
129  };
130 
131  std::string Mean1("Mean_1"); // mean of gaussian
132  std::string Sigma1("Sigma_1"); // sigma of gaussian
133  hydra::Parameter mean1_p = hydra::Parameter::Create().Name(Mean1).Value( 2.5) .Error(0.0001).Limits(0.0, 10.0);
134  hydra::Parameter sigma1_p = hydra::Parameter::Create().Name(Sigma1).Value(0.5).Error(0.0001).Limits(0.01, 1.5);
135 
136  auto gaussian1 = hydra::wrap_lambda(GAUSSIAN1, mean1_p, sigma1_p);

141  (unsigned int npar, const hydra::Parameter* params,unsigned int narg, double* x )
142  {
143  double m2 = (x[0] - params[0])*(x[0] - params[0] );
144  double s2 = params[1]*params[1];
145  double g = exp(-m2/(2.0 * s2 ))/( sqrt(2.0*s2*PI));
146 
147  return g;
148  };
149 
150  std::string Mean2("Mean_2"); // mean of gaussian
