176 inline void Vector4R::applyRotateEuler(GReal_t phi, GReal_t theta, GReal_t ksi)
177 {
178 
179  GReal_t sp = sin(phi);
180  GReal_t st = sin(theta);
181  GReal_t sk = sin(ksi);
182  GReal_t cp = cos(phi);
183  GReal_t ct = cos(theta);
184  GReal_t ck = cos(ksi);
185 
186  GReal_t x = (ck * ct * cp - sk * sp) * v[1]
187  + (-sk * ct * cp - ck * sp) * v[2] + st * cp * v[3];
188  GReal_t y = (ck * ct * sp + sk * cp) * v[1]
189  + (-sk * ct * sp + ck * cp) * v[2] + st * sp * v[3];
190  GReal_t z = -ck * st * v[1] + sk * st * v[2] + ct * v[3];
191 
192  v[1] = x;
193  v[2] = y;
194  v[3] = z;
195 
196 }

214 {
215 
216  GReal_t bx, by, bz, gamma, b2;
217 
218  bx = boost.get(0);
219  by = boost.get(1);
220  bz = boost.get(2);
221 
222  GReal_t bxx = bx * bx;
223  GReal_t byy = by * by;
224  GReal_t bzz = bz * bz;
225 
226  b2 = bxx + byy + bzz;
227 
228  if (b2 > 0.0 && b2 < 1.0)
229  {
230 
231  gamma = 1.0 / sqrt(1.0 - b2);
232 
233  GReal_t gb2 = (gamma - 1.0) / b2;
234 
235  GReal_t gb2xy = gb2 * bx * by;
236  GReal_t gb2xz = gb2 * bx * bz;
237  GReal_t gb2yz = gb2 * by * bz;
238

241  GReal_t gbz = gamma * bz;

245  GReal_t py2 = v[2];
246  GReal_t pz2 = v[3];
247 
248  if (inverse)
249  {
