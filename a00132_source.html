139  __host__ __device__
140  inline ResultVegas operator()( size_t box)
141  {
142 
143  GReal_t volume = 1.0;
144  GReal_t x[NDimensions];
145  volatile GInt_t bin[NDimensions];
146  ResultVegas result;
147  GRND randEng( fSeed+box);
148  thrust::uniform_real_distribution<GReal_t> uniDist(0.0, 1.0);
149 
150  GReal_t m = 0, q = 0;
151  GReal_t f_sq_sum = 0.0;
152 
153  for (size_t call = 0; call < fNCallsPerBox; call++) {
154 
155  for (size_t j = 0; j < NDimensions; j++) {
156 
157  x[j] = uniDist(randEng);
158 
159  GInt_t b = GetBoxCoordinate(box, NDimensions, fNBoxesPerDimension, j);
160 
161  GReal_t z = ((b + x[j]) / fNBoxesPerDimension) * fNBins;
162 
163  GInt_t k = static_cast<GInt_t>(z);
164 
165  GReal_t y = 0;
166  GReal_t bin_width = 0;
167 
168  bin[j] = k;
169 
170  bin_width = fXi[(k + 1)*NDimensions + j]
171  - (k != 0) * fXi[k*NDimensions + j];
172 
173  y = (k != 0) * fXi[k*NDimensions + j] + (z - k) * bin_width;
174 
175  x[j] = fXLow[j] + y * fDeltaX[j];
176 
177  volume *= bin_width;
178  }
179 
180  GReal_t fval = fJacobian*volume*fFunctor( detail::arrayToTuple<GReal_t, NDimensions>(x));
181  GReal_t d = fval - m;
182  m += d / (call + 1.0);
183  q += d * d * (call / (call + 1.0));
184 
185  if (fMode != MODE_STRATIFIED)
186  {
187  for (GUInt_t j = 0; j < NDimensions; j++) {
188 #ifdef __CUDA_ARCH__
189  atomicAdd((fDistribution + bin[j]* NDimensions + j) , static_cast<GFloat_t>(fval*fval));
190 #else
191 #pragma omp atomic
192  *(fDistribution + bin[j]* NDimensions + j) += static_cast<GFloat_t>(fval*fval);
193 #endif
194  }
195  }
196  }
197 
198  result.integral += m*fNCallsPerBox;
199  f_sq_sum = q*fNCallsPerBox;
200  result.tss += f_sq_sum;
201 
202  if (fMode == MODE_STRATIFIED) {
203  for (GUInt_t j = 0; j < NDimensions; j++) {
204 #ifdef __CUDA_ARCH__
205  atomicAdd((fDistribution + bin[j]*NDimensions+j), static_cast<GFloat_t>(f_sq_sum));
206 #else
207 #pragma omp atomic
208  *(fDistribution + bin[j]* NDimensions + j) += static_cast<GReal_t>(f_sq_sum);
209 #endif
210  }
211  }
212 
213  return result;
214 
215  }
