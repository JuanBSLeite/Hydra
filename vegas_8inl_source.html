100  //number of dimensions (user can change it)
101  constexpr size_t N = 5;
102 
103  //integration region limits
104  double min[N];
105  double max[N];
106 
107  //5D Gaussian parameters
108  double mean = 0.0;
109  double sigma = 1.0;
110 
111  //set Gaussian parameters and
112  //integration region limits
113  for(size_t i=0; i< N; i++){
114  min[i] = -6.0;
115  max[i] = 6.0;
116  }
117 
118  // create functor using C++11 lambda
119  auto GAUSSIAN = [=] __host__ __device__ (unsigned int n, double* x ){
120 
121  double g = 1.0;
122  double f = 0.0;
123 
124  for(size_t i=0; i<N; i++){
125 
126  double m2 = (x[i] - mean )*(x[i] - mean );
127  double s2 = sigma*sigma;
128  f = exp(-m2/(2.0 * s2 ))/( sqrt(2.0*s2*PI));
129  g *= f;
130  }
131 
132  return g;
133  };
134 
135  //wrap the lambda
136  auto gaussian = hydra::wrap_lambda(GAUSSIAN);
137 
138  //device

152  //vegas integrator
