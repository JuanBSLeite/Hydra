203 inline void Vector4R::applyRotateEuler(GReal_t phi, GReal_t theta, GReal_t ksi)
204 {
205 
206  GReal_t sp = sin(phi);
207  GReal_t st = sin(theta);
208  GReal_t sk = sin(ksi);
209  GReal_t cp = cos(phi);
210  GReal_t ct = cos(theta);
211  GReal_t ck = cos(ksi);
212 
213  GReal_t x = (ck * ct * cp - sk * sp) * v[1]
214  + (-sk * ct * cp - ck * sp) * v[2] + st * cp * v[3];
215  GReal_t y = (ck * ct * sp + sk * cp) * v[1]
216  + (-sk * ct * sp + ck * cp) * v[2] + st * sp * v[3];
217  GReal_t z = -ck * st * v[1] + sk * st * v[2] + ct * v[3];
218 
219  v[1] = x;
220  v[2] = y;
221  v[3] = z;
222 
223 }

239 __host__ __device__
240 inline void Vector4R::applyBoostTo(const Vector3R& boost, bool inverse)
241 {
242 
243  GReal_t bx, by, bz, gamma, b2;
244 
245  bx = boost.get(0);
246  by = boost.get(1);
247  bz = boost.get(2);
248 
249  GReal_t bxx = bx * bx;
250  GReal_t byy = by * by;
251  GReal_t bzz = bz * bz;
252 
253  b2 = bxx + byy + bzz;
254 
255  if (b2 > 0.0 && b2 < 1.0)
256  {
257 
258  gamma = 1.0 / sqrt(1.0 - b2);
259 
260  GReal_t gb2 = (gamma - 1.0) / b2;
261 
262  GReal_t gb2xy = gb2 * bx * by;
263  GReal_t gb2xz = gb2 * bx * bz;
264  GReal_t gb2yz = gb2 * by * bz;
265 
266  GReal_t gbx = gamma * bx;
267  GReal_t gby = gamma * by;
268  GReal_t gbz = gamma * bz;
269 
270  GReal_t e2 = v[0];
271  GReal_t px2 = v[1];
272  GReal_t py2 = v[2];
273  GReal_t pz2 = v[3];
274 
275  if (inverse)
276  {
277  v[0] = gamma * e2 - gbx * px2 - gby * py2 - gbz * pz2;
278 
279  v[1] = -gbx * e2 + gb2 * bxx * px2 + px2 + gb2xy * py2
280  + gb2xz * pz2;
281 
282  v[2] = -gby * e2 + gb2 * byy * py2 + py2 + gb2xy * px2
283  + gb2yz * pz2;
284 
285  v[3] = -gbz * e2 + gb2 * bzz * pz2 + pz2 + gb2yz * py2
286  + gb2xz * px2;
287  }
288  else
289  {
290  v[0] = gamma * e2 + gbx * px2 + gby * py2 + gbz * pz2;
291 
292  v[1] = gbx * e2 + gb2 * bxx * px2 + px2 + gb2xy * py2 + gb2xz * pz2;
293 
294  v[2] = gby * e2 + gb2 * byy * py2 + py2 + gb2xy * px2 + gb2yz * pz2;
295 
296  v[3] = gbz * e2 + gb2 * bzz * pz2 + pz2 + gb2yz * py2 + gb2xz * px2;
297  }
298  }

361 inline Vector4R Vector4R::cross(const Vector4R& p2)
362 {
363 
364  //Calcs the cross product. Added by djl on July 27, 1995.
365  //Modified for real vectros by ryd Aug 28-96
366 
367  Vector4R temp;
368 
369  temp.v[0] = 0.0;
370  temp.v[1] = v[2] * p2.v[3] - v[3] * p2.v[2];
371  temp.v[2] = v[3] * p2.v[1] - v[1] * p2.v[3];
372  temp.v[3] = v[1] * p2.v[2] - v[2] * p2.v[1];
373 
374  return temp;
375 }
