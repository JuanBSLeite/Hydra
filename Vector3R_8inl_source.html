158 inline void Vector3R::applyRotateEuler(GReal_t phi, GReal_t theta, GReal_t ksi)
159 {
160 
161  GReal_t temp[3];
162  GReal_t sp, st, sk, cp, ct, ck;
163 
164  sp = sin(phi);
165  st = sin(theta);
166  sk = sin(ksi);
167  cp = cos(phi);
168  ct = cos(theta);
169  ck = cos(ksi);
170 
171  temp[0] = (ck * ct * cp - sk * sp) * v[0] + (-sk * ct * cp - ck * sp) * v[1]
172  + st * cp * v[2];
173  temp[1] = (ck * ct * sp + sk * cp) * v[0] + (-sk * ct * sp + ck * cp) * v[1]
174  + st * sp * v[2];
175  temp[2] = -ck * st * v[0] + sk * st * v[1] + ct * v[2];
176 
177  v[0] = temp[0];
178  v[1] = temp[1];
179  v[2] = temp[2];
180 }
