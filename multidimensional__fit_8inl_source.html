121  auto GAUSSIAN = [=] __host__ __device__
122  (unsigned int npar, const hydra::Parameter* params,
123  unsigned int narg, double* x )
124  {
125 
126  double g = 1.0;
127 
128  double mean[3] = {params[0].GetValue(), params[2].GetValue(), params[4].GetValue()};
129  double sigma[3]= {params[1].GetValue(), params[3].GetValue(), params[5].GetValue()};
130 
131  for(size_t i=0; i<narg; i++) {
132 
133  double m2 = (x[i] - mean[i] )*(x[i] - mean[i] );
134  double s2 = sigma[i]*sigma[i];
135  g *= exp(-m2/(2.0 * s2 ))/( sqrt(2.0*s2*PI));
136  }
137 
138  return g;
139  };
