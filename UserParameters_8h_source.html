86  void AddParameter( Parameter* param, GBool_t update_size=1 )
87  {
88  if(param->HasError() && param->IsLimited()){
89  fMnState->Add(param->GetName(), param->GetValue(),
90  param->GetError(), param->GetLowerLim(), param->GetUpperLim());
91  }
92 
93  if(param->HasError() && !param->IsLimited()){
94  fMnState->Add(param->GetName(), param->GetValue(), param->GetError());
95  }
96 
97  if(!param->HasError() && !param->IsLimited()){
98  fMnState->Add(param->GetName(), param->GetValue());
99  }
100 
101  param->SetIndex( fMnState->Index( param->GetName()) );
102 
103  if( update_size)fVariables.push_back(param);
104 
105  }

121  void UpdateParameterLimits(ROOT::Minuit2::MinosError const& minos_error )
122  {
123 
124  std::string name = minos_error.LowerState().Name( minos_error.Parameter());
125  auto parameter = std::find_if(fVariables.begin(), fVariables.end(),
126  [&](Parameter* const p) { return std::string(p->GetName()) == name; } );
127 
128  if (parameter == std::end(fVariables)) {
129  HYDRA_LOG(WARNING, " Parameter :"<< name << " not found. Limits not set.\n\n")
130  return;
131  }
132  (*parameter)->SetLowerLim(minos_error.Lower() );
133  (*parameter)->SetUpperLim(minos_error.Lower() );
134 
135  return;
136  }
