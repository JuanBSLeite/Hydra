67 /**
68  * @class
69  * @ingroup fit
70  * @brief Build a pdf adding other pdfs.
71  *
72  * Given N unnormalized pdfs \f$F_i\f$ , this class define a object representing the sum
73  * \f[ F_t = \sum_i^N c_i \times F_i \f]
74  * The coefficients of the pdfs can represent fractions or yields. If the number of coefficients is equal to
75  * the number of pdfs, the coefficients are interpreted as yields. If the number of coefficients is \f$(N-1)\f$,
76  * the coefficients are interpreted as fractions defined in the interval [0,1].
77  * The coefficient of the last term is calculated as \f$ c_N=1 -\sum_i^{(N-1)} c_i \f$.
78  */
79 template<typename PDF1, typename PDF2, typename ...PDFs>
80 class PDFSumNonExtendable: public detail::AddPdfBase<PDF1,PDF2,PDFs...>

104  /**
105  * \brief Ctor for used to build AddPdf usable in extended likelihood fits.
106  * \param pdf1 first pdf object,
107  * \param pdf2 second pdf object,
108  * \param pdfs remaining pdfs.
109  * \param coef arrary of Parameters, each parameter correspond to a coefficient
110  * \param extend build pdf to be used in extended fit. Default is true.
111  *
112  * Each component pdf is normalized properly before evaluation each time SetParameters(const std::vector<double>& parameters) is called.
113  * The sum is normalized also.
114  */
115  PDFSumNonExtendable( PDF1 const& pdf1, PDF2 const& pdf2, PDFs const& ...pdfs,
116  std::array<Parameter*, npdfs-1>const& coef ):
117  fPDFs(HYDRA_EXTERNAL_NS::thrust::make_tuple(pdf1,pdf2,pdfs...) ),
118  fFunctors(HYDRA_EXTERNAL_NS::thrust::make_tuple(pdf1.GetFunctor(),pdf2.GetFunctor(),pdfs.GetFunctor() ...) ),
119  fCoefSum(1.0),
120  fCoefPartialSum(0.0)
121  {
122  double fraction=1.0;
123  size_t i=0;
124  for(Parameter* var:coef){
125  fCoeficients[i] = *var;
126  fraction -= var->GetValue();
127  fCoefPartialSum += var->GetValue();
128  i++;
129  }
130 
131  fCoeficients[npdfs-1].Name("1.0 - @sum_i{c_i}")
132  .Value(fraction)
133  .Error(0.000)
134  .Limits(0.0, 1.0);
135  }

174  /**
175  * \brief Set the coefficients and parameters of all pdfs.
176  * This method sets the values of all coefficients and parameters of pdfs stored in the PDFSumNonExtendable object.
177  * User should ensure this method is called before the object evaluation.
178  * @param parameters std::vector<double> containing the list of parameters passed by ROOT::Minuit2.
179  */
180  inline void SetParameters(const std::vector<double>& parameters){
181 
182  fCoeficients[ npdfs-1] = 1.0;
183  fCoefPartialSum=0.0;
184  for(size_t i=0; i< npdfs-1; i++){
185  fCoeficients[i].Reset(parameters );
186  fCoeficients[ npdfs-1] -= fCoeficients[i].GetValue();
187  fCoefPartialSum += fCoeficients[i].GetValue();
188  }
189 
190  detail::set_functors_in_tuple(fPDFs, parameters);
191 
192  fFunctors = get_functor_tuple(fPDFs);
193 
194  fCoefSum=0;
195 
196  for(size_t i=0; i< npdfs ; i++)
197  fCoefSum+=fCoeficients[i];
198  }

268  GReal_t operator()(T1&& t )
269  {
270  auto pdf_res_tuple = detail::invoke<pdfs_tuple_type, T1>( t, fPDFs);
271  GReal_t pdf_res_array[npdfs];
272  detail::tupleToArray( pdf_res_tuple, pdf_res_array );
273 
274  GReal_t result = 0;
