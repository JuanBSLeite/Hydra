66  __host__ __device__ inline
67  ResultVegas operator()(ResultVegas const& x, ResultVegas const& y)
68  {
69  ResultVegas result;
70 
71  GReal_t n = x.fN + y.fN;
72 
73  GReal_t delta = y.fMean - x.fMean;
74  GReal_t delta2 = delta * delta;
75 
76  result.fN = n;
77 
78  //result.fMean = x.fMean + delta * y.fN / n;
79  result.fMean = (x.fMean*x.fN + y.fMean*y.fN)/n;
80  result.fM2 = x.fM2 + y.fM2;
81  result.fM2 += delta2 * x.fN * y.fN / n;
82 
83  return result;
84  }

169  __host__ __device__ inline
170  void get_point(const size_t index, GReal_t &volume, GInt_t (&bin)[NDimensions], GReal_t (&x)[NDimensions] )
171  {
172 
173  size_t box = index/fNCallsPerBox;
174 
175  GRND randEng( hash(fSeed,index) );
176  //randEng.discard(index);
177  HYDRA_EXTERNAL_NS::thrust::uniform_real_distribution<GReal_t> uniDist(0.0, 1.0);
178 
179  for (size_t j = 0; j < NDimensions; j++)
180  {
181  x[j] = uniDist(randEng);
182 
183  GInt_t b = fNBoxesPerDimension > 1? GetBoxCoordinate(box, NDimensions, fNBoxesPerDimension, j):box;
184 
185  GReal_t z = ((b + x[j]) / fNBoxesPerDimension) * fNBins;
186 
187  GInt_t k = static_cast<GInt_t>(z);
188 
189  GReal_t y = 0;
190  GReal_t bin_width = 0;
191 
192  bin[j] = k;
193 
194  bin_width = fXi[(k + 1)*NDimensions + j]
195  - (k != 0) * fXi[k*NDimensions + j];
196 
197  y = (k != 0) * fXi[k*NDimensions + j] + (z - k) * bin_width;
198 
199  x[j] = fXLow[j] + y * fDeltaX[j];
200 
201  volume *= bin_width;
202 
203  // printf("j=%d k=%d z=%f y=%f box=%f x[j]=%f \n", j, k, z,y, double(b) , x[j] );
204 }
