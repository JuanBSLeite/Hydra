284  GULong_t Unweight(size_t seed)
285  {
286  /**
287  * Flag the accepted and rejected events
288  */
289 
290 
291 
292  GULong_t count = 0;
293  if(N==2)
294  {
295  thrust::fill(fFlags.begin(), fFlags.end(), kTrue);
296  count = fNEvents;
297  }
298  else
299  {
300 
301  auto w = thrust::max_element(fWeights.begin(),fWeights.end());
302  fMaxWeight=*w;
303  // create iterators
304  thrust::counting_iterator<GLong_t> first(0);
305  thrust::counting_iterator<GLong_t> last = first + fNEvents;
306 
307 
308  thrust::transform(first, last, fWeights.begin(),
309  fFlags.begin(), detail::FlagAcceptReject(seed, fMaxWeight));
310 
311  count = thrust::count(fFlags.begin(), fFlags.end(),
312  kTrue);
313 
314  }
315 
316  return count;
317 
318  }
