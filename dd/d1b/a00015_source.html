70 struct Gauss:public BaseFunctor< Gauss<N>, GReal_t, 0>
71 {
72  Gauss(){};
73  Gauss(std::array<GReal_t, N> const& means, std::array<GReal_t, N> const& sigmas ) {
74  for(size_t i=0;i<N;i++) {
75 
76  fM[i] = means[i];
77  fS[i] = sigmas[i];
78  }
79  }

81  __host__ __device__
82  inline Gauss(Gauss<N> const& other) {
83  for(size_t i=0;i<N;i++) {
84 
85  fM[i] = other.fM[i];
86  fS[i] = other.fS[i];
87  }
88  }

90  template<typename T>
91  __host__ __device__
92  inline GReal_t Evaluate(T* X)
93  {
94  GReal_t gi=1.0;
95  for (size_t i = 0; i < N; i++){
96 
97  GReal_t x = X[i];
98  gi *= exp(-((x - fM[i]) * (x - fM[i])) / (2 * fS[i] * fS[i]));
99  // / (sqrt(2.0 * 3.14159265358979323846) * fS[i]);
100  }
101 
102  return gi;
103  }
104 
105  GReal_t fM[N];
106  GReal_t fS[N];
107 };
