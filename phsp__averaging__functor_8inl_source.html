132  //C++11 lambda for cosine of helicity angle Kpi
133  auto COSHELANG = [] __host__ __device__ (unsigned int n, hydra::Vector4R *fvectors )
134  {
135  hydra::Vector4R p1 = fvectors[0];
136  hydra::Vector4R p2 = fvectors[1];
137  hydra::Vector4R p3 = fvectors[2];
138 
139  hydra::Vector4R p = p1 + p2 + p3;
140  hydra::Vector4R q = p2 + p3;
141 
142  double pd = p * p2;
143  double pq = p * q;
144  double qd = q * p2;
145  double mp2 = p.mass2();
146  double mq2 = q.mass2();
147  double md2 = p2.mass2();
148 
149  return (pd * mq2 - pq * qd)
150  / sqrt((pq * pq - mq2 * mp2) * (qd * qd - mq2 * md2));
151  };
152 
153  auto cosTheta = hydra::wrap_lambda(COSHELANG);
154 
155  hydra::Vector4R B0(B0_mass, 0.0, 0.0, 0.0);
156  double masses[3]{Jpsi_mass, K_mass, pi_mass };
157 
158  // Create PhaseSpace object for B0-> K pi J/psi
159  hydra::PhaseSpace<3> phsp(masses);
