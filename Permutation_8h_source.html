50 template <class Iterator>
51 __host__ __device__
52 Iterator rotate(Iterator first, Iterator n_first, Iterator last)
53 {
54 
55  using hydra::detail::iter_swap;
56 
57  if(first == n_first) return last;
58  if(n_first == last) return first;
59 
60  Iterator next = n_first;
61 
62  do {
63  iter_swap(first++, next++);
64  if (first == n_first) n_first = next;
65  }
66  while (next != last);
67 
68  Iterator ret = first;
69 
70  for(next = n_first; next != last; ) {

72  if(first == n_first) n_first = next;
73  else if(next == last) next = n_first;
74  }
75

79 template<typename Iterator, typename Integer, typename Comparator>
80 typename std::enable_if<std::is_integral<Integer>::value, void>::type
81 __host__ __device__
82 nth_permutation(Iterator begin, Iterator end, Integer n, Comparator comp)
83 {
84  HYDRA_EXTERNAL_NS::thrust::sort(begin, end, comp);
85 
86  const Integer k(end - begin);
87 
88  Integer radix = Integer(1);
89  for(Integer i = Integer(2); i < k; ++i)
90  radix *= i;
91 
92  for(Integer i = 0; i < k; ++i)
93  {
94 
95  Integer digit = n / radix;
96  rotate(begin, begin + digit, begin + digit + 1);
97  ++ begin;
98  n %= radix;
99  if (i + 1 != k)
100  radix /= (k - i - 1);
101  }
102 }

110 bool next_permutation(BidirIt first, BidirIt last)
111 {
112  if (first == last) return false;
113  BidirIt i = last;
114  if (first == --i) return false;
115 
116  while (true) {
