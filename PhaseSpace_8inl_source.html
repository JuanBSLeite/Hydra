266 template <size_t N, typename GRND>
267 inline GReal_t PhaseSpace<N,GRND>::PDK(const GReal_t a, const GReal_t b, const GReal_t c) const {
268  //the PDK function
269  GReal_t x = (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c);
270  x = sqrt(x) / (2 * a);
271  return x;
272 }
