102 // ProcessCallsPlainBinary is a functor that accepts two PlainState
103 // structs and returns a new summary_stats_data which are an
104 // approximation to the summary_stats for
105 // all values that have been agregated so far
106 struct ProcessCallsPlainBinary
107  : public thrust::binary_function<PlainState const&,
108  PlainState const&,
109  PlainState >
110 {
111  __host__ __device__ inline
112  PlainState operator()(const PlainState& x, const PlainState& y)
113  {
114  PlainState result;
115 
116  // precompute some common subexpressions
117  size_t n = x.fN + y.fN;
118 
119  GReal_t delta = y.fMean - x.fMean;
120  GReal_t delta2 = delta * delta;
121 
122  //Basic number of samples (n), min, and max
123  result.fN = n;
124  result.fMin = thrust::min(x.fMin, y.fMin);
125  result.fMax = thrust::max(x.fMax, y.fMax);
126 
127  result.fMean = x.fMean + delta * y.fN / n;
128 
129  result.fM2 = x.fM2 + y.fM2;
130  result.fM2 += delta2 * x.fN * y.fN / n;
131 
132  return result;
133  }
