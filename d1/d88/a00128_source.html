385 inline void Vector4R::applyBoostTo(const Vector3R& boost, bool inverse)
386 {
387 
388  GReal_t bx, by, bz, gamma, b2;
389 
390  bx = boost.get(0);
391  by = boost.get(1);
392  bz = boost.get(2);
393 
394  GReal_t bxx = bx * bx;
395  GReal_t byy = by * by;
396  GReal_t bzz = bz * bz;
397 
398  b2 = bxx + byy + bzz;
399 
400  if (b2 > 0.0 && b2 < 1.0)
401  {
402 
403  gamma = 1.0 / sqrt(1.0 - b2);
404 
405  GReal_t gb2 = (gamma - 1.0) / b2;
406 
407  GReal_t gb2xy = gb2 * bx * by;
408  GReal_t gb2xz = gb2 * bx * bz;
409  GReal_t gb2yz = gb2 * by * bz;
410 
411  GReal_t gbx = gamma * bx;
412  GReal_t gby = gamma * by;
413  GReal_t gbz = gamma * bz;
414 
415  GReal_t e2 = v[0];
416  GReal_t px2 = v[1];
417  GReal_t py2 = v[2];
418  GReal_t pz2 = v[3];
419 
420  if (inverse)
421  {
422  v[0] = gamma * e2 - gbx * px2 - gby * py2 - gbz * pz2;
423 
424  v[1] = -gbx * e2 + gb2 * bxx * px2 + px2 + gb2xy * py2
425  + gb2xz * pz2;
426 
427  v[2] = -gby * e2 + gb2 * byy * py2 + py2 + gb2xy * px2
428  + gb2yz * pz2;
429 
430  v[3] = -gbz * e2 + gb2 * bzz * pz2 + pz2 + gb2yz * py2
431  + gb2xz * px2;
432  }
433  else
434  {
435  v[0] = gamma * e2 + gbx * px2 + gby * py2 + gbz * pz2;
436
