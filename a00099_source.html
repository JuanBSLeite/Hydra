81  __host__ __device__ void bbsort(GReal_t *array, GInt_t n)
82  {
83  // Improved bubble sort
84  for (GInt_t c = 0; c < n; c++)
85  {
86  GInt_t nswap = 0;
87 
88  for (GInt_t d = 0; d < n - c - 1; d++)
89  {
90  if (array[d] > array[d + 1]) /* For decreasing order use < */
91  {
92  GReal_t swap = array[d];
93  array[d] = array[d + 1];
94  array[d + 1] = swap;
95  nswap++;
96  }
97  }
98  if (nswap == 0)
99  break;
100  }
