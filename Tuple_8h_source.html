39 /*! \p tuple template is an alias to the variadic version of HYDRA_EXTERNAL_NS::thrust::tuple
40  * and that can be instantiated with a indefinite number of arguments.
41  * Each template argument specifies the type of element in the \p tuple.
42  * Consequently, tuples are heterogeneous, fixed-size collections of values. An
43  * instantiation of \p tuple with two arguments is similar to an instantiation
44  * of \p pair with the same two arguments. Individual elements of a \p tuple may
45  * be accessed with the \p get function.
46  *
47  * \tparam TN The type of the <tt>N</tt> \c tuple element.
48  *
49  * The following code snippet demonstrates how to create a new \p tuple object
50  * and inspect and modify the value of its elements.
51  *
52  * \code
53  * #include <hydra/tuple.h>
54  * #include <iostream>
55  * ...
56  * // create a tuple containing an int, a float, and a string
57  * hydra::tuple<int, float, const char*> t(13, 0.1f, "hydra");
58  *
59  * // individual members are accessed with the free function get
60  * std::cout << "The first element's value is " << hydra::get<0>(t) << std::endl;
61  *
62  * // or the member function get
63  * std::cout << "The second element's value is " << t.get<1>() << std::endl;
64  *
65  * // we can also modify elements with the same function
66  * hydra::get<0>(t) += 10;
67  * \endcode
68  */
69 template<typename... T> using tuple = HYDRA_EXTERNAL_NS::thrust::tuple<T...>;
