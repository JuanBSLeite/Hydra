156  //C++11 lambda for invariant mass
157  auto M2 = [] __host__ __device__ (hydra::Vector4R const& p1, hydra::Vector4R const& p2 )
158  { return ( p1 + p2).mass2(); };
159 
160 
161  //C++11 lambda for cosine of helicity angle Kpi
162  auto COSHELANG = [] __host__ __device__ (hydra::Vector4R const& p1, hydra::Vector4R const& p2, hydra::Vector4R const& p3 )
163  {
164  hydra::Vector4R p = p1 + p2 + p3;
165  hydra::Vector4R q = p2 + p3;
166 
167  double pd = p * p2;
168  double pq = p * q;
169  double qd = q * p2;
170  double mp2 = p.mass2();
171  double mq2 = q.mass2();
172  double md2 = p2.mass2();
173 
174  return (pd * mq2 - pq * qd)
175  / sqrt((pq * pq - mq2 * mp2) * (qd * qd - mq2 * md2));
176  };
177 
178  //C++11 lambda for angle between the planes [K,pi] and [mu+, mu-]
179  auto DELTA = [] __host__ __device__ (hydra::Vector4R const& d2, hydra::Vector4R const& d3,
180  hydra::Vector4R const& h1, hydra::Vector4R const& h2 )
181  {
182  hydra::Vector4R D = d2 + d3;
183 
184  hydra::Vector4R d1_perp = d2 - (D.dot(d2) / D.dot(D)) * D;
185  hydra::Vector4R h1_perp = h1 - (D.dot(h1) / D.dot(D)) * D;
186 
187  // orthogonal to both D and d1_perp
188  hydra::Vector4R d1_prime = D.cross(d1_perp);
189 
190  d1_perp = d1_perp / d1_perp.d3mag();
191  d1_prime = d1_prime / d1_prime.d3mag();
192 
193  double x, y;
194 
195  x = d1_perp.dot(h1_perp);
196  y = d1_prime.dot(h1_perp);
197 
198  double chi = atan2(y, x);
199 
200  if(chi < 0.0) chi += 2.0*PI;
201 
202  return chi;
203  };
